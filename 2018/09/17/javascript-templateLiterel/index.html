파라미터는 입력되는 정보의 형식이 있다
아규먼트는 그 형식에 맞게 실제로 입력한 값을 argument라고 한다

function sum(num1, num2){ // 매개변수 parameter라고한다
console.log(num1 + num2);
}

sum(2,4); // 각각의 입력값 하나하나를 인자(argument)라고 한다.

sum(2,4); // 그러나 우리가 만든 함수 sum은 융통성이 없다 언제나 console.log로 화면을 출력한다

sum(2,Math.round(1.4)); //값: 3 반면 Math.round라는 함수는 값 자체로 출력되기에 융통성이 있고 응용이 가능하다.

function sum(num1, num2){
return num1 + num2); // 이런식으로 return으로 바꿈으로써 융통성있게 값자체를 전달하는 함수가 되었다
console.log(‘return은 함수의 종료를 의미하기도 한다.’); // 이곳은 출력되지 않는다.
}

// 함수는 처리방법을 가지고 있는 일종의 구문(statement)이면서 동시에 값(value)이다.
//
// 함수는 for나 while같은 구문과 달리 변수에 넣을 수 있고 값이 될 수 있다.

var a = [f]; // f가 담긴 배열을 변수 a에 담았다

a0; // a의 첫번째 인덱스에는 f라는 함수가 담겨있고 실행가능하다 즉 배열의 원소로서 함수가 들어갈수있다.

var o = {func:f} // o라는 객체를 만들고 프로퍼티로 func:f 로 저장하고 호출하면?
o.func(); // 마찬가지로 실행가능한 함수가 된다.

// 처리방법을 기술해 그룹핑하는 함수는 데이터이기도 하기때문에 배열과 객체에다 담을 수 있다.

// 객체에는 주로 함수를 담는다 왜냐면 이름으로 담아 사용할 수 있기때문이다.

var v1 = ‘v1’;
var v2 = ‘v2’;
// 수많은 변수들이 늘비하고 프로젝트의 규모가 커지게 되면
// v1과 중복되는 변수가 존재할 가능성이 커진다.

// 이렇게 객체라는 바구니에 그룹핑해 담아 둠으로써 변수를 겹치게 사용하지 않을 수 있다
var o = {
v1:’v1’,
v2:’v2’,
f1:function(){
console.log(o.v1);
},
f2:function(){
console.log(o.v2);
}
}
o.f1(); // 문제없이 실행된다
o.f2();

// 하지만!! 이러한 형태도 문제가 있다
var p = {
// 이 함수는 자신이 속해있는 객체를 참조해 값을 출력하는 함수였기 때문에
// p로 이름이 바뀔경우 객체를 참조할 수 없고 그것보다
v1:’v1’,
v2:’v2’,
f1:function(){
console.log(o.v1);
// 만약 o라는 객체가 바깥에 존재할 경우 o라는 객체를 참조해 엉뚱한 결과를
// 나타낼 수도 있다.
// 자기자신을 참조해 값을 출력하려 한 함수가 엉뚱한 결과를 나타내게 되는 것이다.
},
f2:function(){
console.log(o.v2);
}
}
p.f1(); // Error!!
p.f2(); // p라는 변수로 설정해두면 참조할수가 없게된다

// 그게 바로 this다
var p = {

v2:’v2’,
f1:function(){
console.log(this.v1); // this라는 키워드를 사용함으로써 자신이 속해있는 객체를 가리키고 있어
},
f2:function(){
console.log(this.v2);
}
}
p.f1(); // 어떤 이름의 객체에 담겨있는지 상관없이 자기자신이 속한 객체를 가리키므로 참조할 수 있게된다.
p.f2();

// An object can be passed as the first argument to call or apply and this will be bound to it.
var obj = {a: ‘Custom’};

// This property is set on the global object
var a = ‘Global’;

function whatsThis() {
return this.a; // The value of this is dependent on how the function is called [ 이 값은 함수 호출 방법에 따라 다릅니다.]
}

whatsThis(); // ‘Global’
whatsThis.call(obj); // ‘Custom’
whatsThis.apply(obj); // ‘Custom’

function add(c, d) { [ 함수 add (c, d) {]
return this.a + this.b + c + d; [ this.a + this.b + c + d를 반환하십시오.]
} [ }]

var o = {a: 1, b: 3}; [ var o = {a : 1, b : 3};]

// The first parameter is the object to use as [ // 첫 번째 매개 변수는 다음과 같이 사용할 객체입니다.]
// ‘this’, subsequent parameters are passed as [ // ‘this’, 후속 매개 변수는 다음과 같이 전달됩니다.]
// arguments in the function call [ // 함수 호출의 인수]
add.call(o, 5, 7); // 16 [ add.call (o, 5, 7); // 16]

// The first parameter is the object to use as [ // 첫 번째 매개 변수는 다음과 같이 사용할 객체입니다.]
// ‘this’, the second is an array whose [ // ‘this’, 두 번째 배열은]
// members are used as the arguments in the function call [ // 멤버는 함수 호출에서 인수로 사용됩니다.]
add.apply(o, [10, 20]); // 34 [ add.apply (o, [10, 20]); // 34]

console.log(Object.prototype.toString.call(this));
